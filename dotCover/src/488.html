using Microsoft.AspNetCore.Mvc;
using VirtoCommerce.Storefront.Model;
using VirtoCommerce.Storefront.Model.Common;

namespace VirtoCommerce.Storefront.Controllers
{
    public class MarketingController : StorefrontControllerBase
    {
        public MarketingController(IWorkContextAccessor workContextAccessor, IStorefrontUrlBuilder urlBuilder)
            : base(workContextAccessor, urlBuilder)
        {
        }

        // GET: /trends
        [HttpGet]
        public ActionResult Trends()
        {
            WorkContext.CurrentPageSeo = new SeoInfo { Title = "Тренды" };

            return View("trends");
        }

        // GET: /trends/{id}
        [HttpGet]
        public ActionResult Trend(string id)
        {
            WorkContext.CurrentPageSeo = new SeoInfo { Title = "Тренды" };

            return View("trend");
        }

        // GET: /promotions
        [HttpGet]
        public ActionResult Promotions()
        {
            WorkContext.CurrentPageSeo = new SeoInfo { Title = "Акции" };

            return View("promotions");
        }

        // GET: /promotions/{id}
        [HttpGet]
        public ActionResult Promotion(string id)
        {
            WorkContext.CurrentPageSeo = new SeoInfo { Title = "Акции" };

            return View("promotion");
        }
    }
}
