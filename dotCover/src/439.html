using Microsoft.AspNetCore.Authorization;
using VirtoCommerce.Storefront.Model;
using VirtoCommerce.Storefront.Model.Security;

namespace VirtoCommerce.Storefront.Domain.Security
{
    public class OnlyCustomerAuthorizationRequirement : IAuthorizationRequirement
    {
        public const string PolicyName = "OnlyCustomer";
    }

    public class OnlyCustomerAuthorizationHandler : UserTypeAuthorizationHandlerBase<OnlyCustomerAuthorizationRequirement>
    {
        public OnlyCustomerAuthorizationHandler(IWorkContextAccessor workContextAccessor) : base(workContextAccessor)
        {
            UserTypeName = UserType.Customer;
        }
    }
}
