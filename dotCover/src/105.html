using System.Linq;
using VirtoCommerce.LiquidThemeEngine.Objects;
namespace VirtoCommerce.LiquidThemeEngine.Converters
{
    public partial class ShopifyModelConverter
    {

        public virtual ProductSearchCriteria ToLiquidProductSearchCriteria(Storefront.Model.Catalog.ProductSearchCriteria productSearchCriteria)
        {
            var retVal = new ProductSearchCriteria();

            retVal.PageNumber = productSearchCriteria.PageNumber;
            retVal.PageSize = productSearchCriteria.PageSize;

            retVal.SortBy = productSearchCriteria.SortBy;
            retVal.Keyword = productSearchCriteria.Keyword;

            if (productSearchCriteria.Terms.Any())
            {
                retVal.CurrentTags = new TagCollection(productSearchCriteria.Terms.Select(t => ToLiquidTag(t)).ToList());
            }

            return retVal;
        }
    }
}
