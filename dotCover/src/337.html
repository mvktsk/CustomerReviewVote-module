using Microsoft.AspNetCore.Mvc.ApplicationModels;
using Microsoft.AspNetCore.Mvc.Authorization;
using VirtoCommerce.Storefront.Controllers;

namespace VirtoCommerce.Storefront.Infrastructure
{
    public class RequireAuthenticatedUserControllerModelConvention : IControllerModelConvention
    {
        public void Apply(ControllerModel controller)
        {
            if (controller.ControllerType != typeof(AssetController))
            {
                controller.Filters.Add(new AuthorizeFilter("RequireAuthenticatedUser"));
            }
        }
    }
}
