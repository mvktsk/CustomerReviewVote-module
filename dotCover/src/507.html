using System.Threading.Tasks;
using Microsoft.AspNetCore.Mvc;
using VirtoCommerce.Storefront.Extensions;
using VirtoCommerce.Storefront.Model;
using VirtoCommerce.Storefront.Model.Common;
using VirtoCommerce.Storefront.Model.Invoices.Services;

namespace VirtoCommerce.Storefront.Controllers.Api
{
    public class ApiInvoicesController : StorefrontControllerBase
    {
        private readonly IHruInvoicesService _invoicesService;

        public ApiInvoicesController(IWorkContextAccessor workContextAccessor, IStorefrontUrlBuilder urlBuilder, IHruInvoicesService invoicesService)
         : base(workContextAccessor, urlBuilder)
        {
            _invoicesService = invoicesService;
        }

        // GET: storefrontapi/invoices
        [HttpGet]
        [ValidateAntiForgeryToken]
        public async Task<ActionResult> GetInvoices(string salePointId, string beginDate, string endDate)
        {
            var invoices = await _invoicesService.GetInvoices(salePointId, beginDate.ParseMillisecondsToDate(), endDate.ParseMillisecondsToDate()).ConfigureAwait(false);
            
            return Ok(invoices);
        }
    }
}
