using System.Collections.Generic;
using System.Globalization;
using System.Linq;
using VirtoCommerce.Storefront.Model.Common;

namespace VirtoCommerce.Storefront.Model.Catalog
{
    public static class CatalogPropertiesExtensions
    {
        public static string GetValue(this IEnumerable<CatalogProperty> properties, string propertyName)
        {
            string result = null;

            if (properties != null)
            {
                var property = properties.FirstOrDefault(p => p.Name.EqualsInvariant(propertyName));
                if (property != null)
                {
                    result = property.Value;
                }
            }

            return result;
        }

        public static decimal? GetDecimalValue(this IEnumerable<CatalogProperty> properties, string propertyName)
        {
            decimal? result = null;

            var property = properties?.FirstOrDefault(p => p.Name.EqualsInvariant(propertyName));
            if (property != null && !property.Value.IsNullOrEmpty() && decimal.TryParse(property.Value, NumberStyles.Float, CultureInfo.InvariantCulture, out var value))
            {
                result = value;
            }

            return result;
        }

        public static int? GetIntegerValue(this IEnumerable<CatalogProperty> properties, string propertyName)
        {
            int? result = null;

            var property = properties?.FirstOrDefault(p => p.Name.EqualsInvariant(propertyName));
            if (property != null && !property.Value.IsNullOrEmpty() && int.TryParse(property.Value, NumberStyles.Integer, CultureInfo.InvariantCulture, out var value))
            {
                result = value;
            }

            return result;
        }
    }
}
