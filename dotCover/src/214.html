using System;
using Newtonsoft.Json;

namespace VirtoCommerce.Storefront.Model.Invoices
{
    public class HruInvoicesItem
    {
        [JsonProperty("SalepointId")]
        public string SalePointId { get; set; }

        [JsonProperty("ShipmentDate")]
        public DateTime ShipmentDate { get; set; }

        [JsonProperty("PaymentDate")]
        public DateTime PaymentDate { get; set; }

        [JsonProperty("PaymentDays")]
        public int PaymentDays { get; set; }

        [JsonProperty("InvoiceDate")]
        public DateTime InvoiceDate { get; set; }

        [JsonProperty("InvoiceNo")]
        public string InvoiceNo { get; set; }

        [JsonProperty("JeansOrderId")]
        public string JeansOrderId { get; set; }

        [JsonProperty("DistrInvoiceNo")]
        public string DistrInvoiceNo { get; set; }

        [JsonProperty("InvoiceAmount")]
        public decimal InvoiceAmount { get; set; }

        [JsonProperty("InvoiceDebt")]
        public decimal InvoiceDebt { get; set; }

        [JsonProperty("OrderNumber")]
        public string OrderNumber { get; set; }
    }
}
