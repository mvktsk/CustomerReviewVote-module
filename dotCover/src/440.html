using Microsoft.AspNetCore.Authorization;
using VirtoCommerce.Storefront.Model;
using VirtoCommerce.Storefront.Model.Security;

namespace VirtoCommerce.Storefront.Domain.Security
{
    public class OnlyDispatcherAuthorizationRequirement : IAuthorizationRequirement
    {
        public const string PolicyName = "OnlyDispatcher";
    }

    public class OnlyDispatcherAuthorizationHandler : UserTypeAuthorizationHandlerBase<OnlyDispatcherAuthorizationRequirement>
    {
        public OnlyDispatcherAuthorizationHandler(IWorkContextAccessor workContextAccessor) : base(workContextAccessor)
        {
            UserTypeName = UserType.Dispatcher;
        }
    }
}
