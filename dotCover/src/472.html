using Microsoft.Extensions.Options;
using VirtoCommerce.Storefront.Infrastructure;
using VirtoCommerce.Storefront.Model;
using VirtoCommerce.Storefront.Model.Cart.Services;
using VirtoCommerce.Storefront.Model.Common;

namespace VirtoCommerce.Storefront.Domain.Cart
{
    public class HapTimeZoneService : IHapTimeZoneService
    {
        private readonly IWorkContextAccessor _workContextAccessor;
        private readonly StorefrontOptions _options;

        public HapTimeZoneService(IOptions<StorefrontOptions> options, IWorkContextAccessor workContextAccessor)
        {
            _workContextAccessor = workContextAccessor;
            _options = options.Value;
        }

        public string GetTimeZoneId()
        {
            var result = _options.TimeZoneId;

            if (result.IsNullOrEmpty())
            {
                // There should be a string with format like +02:00, -07:00, etc.
                result = _workContextAccessor.WorkContext.CurrentStore.DynamicProperties.GetDynamicPropertyValue("Time Zone ID");
            }

            if (result.IsNullOrEmpty())
            {
                result = "-01:00";
            }

            return result;
        }
    }
}
