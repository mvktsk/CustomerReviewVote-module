using Microsoft.AspNetCore.Authorization;
using Microsoft.AspNetCore.Mvc;
using VirtoCommerce.Storefront.Domain.Security;
using VirtoCommerce.Storefront.Model;
using VirtoCommerce.Storefront.Model.Common;
using VirtoCommerce.Storefront.Model.Order;

namespace VirtoCommerce.Storefront.Controllers
{
    [Authorize(OnlyDistributorAuthorizationRequirement.PolicyName)]
    public class DistributorController : StorefrontControllerBase
    {
        public DistributorController(IWorkContextAccessor workContextAccessor, IStorefrontUrlBuilder urlBuilder)
            : base(workContextAccessor, urlBuilder)
        {
        }

        [HttpGet]
        public ActionResult NewOrders()
        {
            WorkContext.CurrentPageSeo = new SeoInfo { Title = "Distributor New Orders" };

            return View("distributors/new-orders", WorkContext);
        }

        [HttpGet]
        public ActionResult PreviousOrders()
        {
            WorkContext.CurrentPageSeo = new SeoInfo { Title = "Distributor Previous Orders" };

            return View("distributors/previous-orders", WorkContext);
        }

        [HttpGet]
        [Authorize(FullDistributorAuthorizationRequirement.PolicyName)]
        public ActionResult Products()
        {
            WorkContext.CurrentPageSeo = new SeoInfo { Title = "Distributor My Catalogue" };

            return View("distributors/products", WorkContext);
        }

        [HttpGet]
        [AllowAnonymous]
        public ActionResult GetOrderDetails(string orderNumber, string key)
        {
            WorkContext.CurrentPageSeo = new SeoInfo { Title = "Distributor Order" };

            WorkContext.CurrentOrder = new CustomerOrder(WorkContext.CurrentCurrency)
            {
                Number = orderNumber,
                CancelReason = key,
            };

            return View("distributors/order", WorkContext);
        }
    }
}
