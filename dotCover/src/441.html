using Microsoft.AspNetCore.Authorization;
using VirtoCommerce.Storefront.Model;
using VirtoCommerce.Storefront.Model.Security;

namespace VirtoCommerce.Storefront.Domain.Security
{
    public class OnlyDistributorAuthorizationRequirement : IAuthorizationRequirement
    {
        public const string PolicyName = "OnlyDistributor";
    }

    public class OnlyDistributorAuthorizationHandler : UserTypeAuthorizationHandlerBase<OnlyDistributorAuthorizationRequirement>
    {
        public OnlyDistributorAuthorizationHandler(IWorkContextAccessor workContextAccessor) : base(workContextAccessor)
        {
            UserTypeName = UserType.Distributor;
        }
    }
}
