using VirtoCommerce.Storefront.Model.Common.Specifications;

namespace VirtoCommerce.Storefront.Model.Catalog.Specifications
{
    public class NonStandardGiftsIsInStockSpecification : ISpecification<Product>
    {
        public bool IsSatisfiedBy(Product product)
        {
            // non-standart gifts always available for users.
            return product.Sku.ToLower().StartsWith("non-standard");
        }
    }
}
