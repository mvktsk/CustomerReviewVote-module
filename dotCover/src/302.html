using System;
using System.Collections.Generic;

namespace VirtoCommerce.Storefront.Model.Common
{
    public static class DaysEnumerator
    {
        public static IEnumerable<DateTime> Enumerate(DateTime startDate, DateTime endDate)
        {
            for (var day = startDate; day <= endDate; day = day.AddDays(1))
            {
                yield return day;
            }
        }
    }
}
