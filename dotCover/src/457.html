using VirtoCommerce.Storefront.Model.Common;
using VirtoCommerce.Storefront.Model.Marketing;
using marketingDto = VirtoCommerce.Storefront.AutoRestClients.HruMarketingModuleApi.Models;

namespace VirtoCommerce.Storefront.Domain.Marketing
{
    public static class HruCarouselBannerConverter
    {
        public static CarouselBanner ToCarouselBanner(this marketingDto.CarouselBanner carouselBannerDto, IStorefrontUrlBuilder urlBuilder)
        {
            var carouselBanner = new CarouselBanner
            {
                BackgroundImageUrl = carouselBannerDto.BackgroundImageUrl,
                ButtonLink = urlBuilder.ToAppAbsolute(carouselBannerDto.ButtonLink),
                IsActive = carouselBannerDto.IsActive ?? false,
                OverlayImageUrl = carouselBannerDto.OverlayImageUrl,
                Title = carouselBannerDto.Title,
                Subtitle = carouselBannerDto.Subtitle,
                Label = carouselBannerDto.Label,
                HasOwner = !carouselBannerDto.ObjectId.IsNullOrEmpty(),
                ButtonTitle = carouselBannerDto.ButtonTitle
            };

            return carouselBanner;
        }
    }
}
