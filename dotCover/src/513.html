using System.Linq;
using Microsoft.AspNetCore.Http;
using Microsoft.AspNetCore.Mvc;
using VirtoCommerce.Storefront.Domain.Customer;
using VirtoCommerce.Storefront.Model;
using VirtoCommerce.Storefront.Model.Common;

namespace VirtoCommerce.Storefront.Controllers.Api
{
    public class ApiOutletsController : StorefrontControllerBase
    {
        public ApiOutletsController(IWorkContextAccessor workContextAccessor, IStorefrontUrlBuilder urlBuilder)
            : base(workContextAccessor, urlBuilder)
        {
        }

        // POST: storefrontapi/outlets/current/{outletId}
        [HttpPost]
        [ValidateAntiForgeryToken]
        public ActionResult ChangeOutlet(string outletId)
        {
            var outlet = WorkContext?.CurrentUser?.Contact?.Outlets?.FirstOrDefault(o => o.Id.EqualsInvariant(outletId));
            if (outlet != null)
            {
                Response.Cookies.Append(HruCookies.CurrentOutletId, outlet.Id, new CookieOptions { IsEssential = true });
            }

            return Ok();
        }
    }
}
