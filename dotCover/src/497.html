using Microsoft.AspNetCore.Mvc;
using VirtoCommerce.Storefront.Model;
using VirtoCommerce.Storefront.Model.Common;

namespace VirtoCommerce.Storefront.Controllers
{
    public class TemplatesController : StorefrontControllerBase
    {

        public TemplatesController(IWorkContextAccessor workContextAccessor, IStorefrontUrlBuilder urlBuilder)
            : base(workContextAccessor, urlBuilder)
        {
        }

        [HttpGet]
        public ActionResult GetTemplates()
        {
            WorkContext.CurrentPageSeo = new SeoInfo { Title = "Шаблоны заказов" };

            return View("templates");
        }
    }
}
