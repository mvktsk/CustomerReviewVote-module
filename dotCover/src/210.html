using System.Collections.Generic;
using VirtoCommerce.Storefront.Model.Common;

namespace VirtoCommerce.Storefront.Model.Marketing
{
    /// <summary>
    /// Represents promotion object
    /// </summary>
    public partial class Promotion : Entity
    {
        public IList<string> Coupons { get; set; }
        public string Description { get; set; }
        public string Name { get; set; }
        public bool UsePromoSku { get; set; }
        public string Type { get; set; }

        public string GetReferenceCode()
        {
            string result = null;

            // Return the first line of the description
            var lines = Description?.Split('\n', 2);
            if (lines?.Length > 0)
            {
                result = lines[0].Trim();
            }

            return result;
        }
    }
}
